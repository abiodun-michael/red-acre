import mongoose from "mongoose";

const {Schema} = mongoose

/**
 * Transaction Schema
 */
const TransactionSchema = new Schema({
    amount1: {
        type:Number,
        required:true
    },
    currencyFrom:{
        type:String,
        required:true
    },
    amount2:{
        type:Number,
        required:true
    },
    currencyTo:{
        type:String,
        required:true
    },
    createdAt:{
        type: Date, 
        default: Date.now,
        required:true
    },
    extType:{
        type:String,
        enum:["LIVE","EXCHANGE"],
        default:"LIVE",
        required:true
    }
})


const TransactionModel = mongoose.model("Transaction", TransactionSchema)

export default TransactionModel
